00055 // This structure needs to be passed to mg_start(), to let mongoose know
00056 // which callbacks to invoke. For detailed description, see
00057 // https://github.com/valenok/mongoose/blob/master/UserManual.md
00058 struct mg_callbacks {
00059   // Called when mongoose has received new HTTP request.
00060   // If callback returns non-zero,
00061   // callback must process the request by sending valid HTTP headers and body,
00062   // and mongoose will not do any further processing.
00063   // If callback returns 0, mongoose processes the request itself. In this case,
00064   // callback must not send any data to the client.
00065   int  (*begin_request)(struct mg_connection *);
00066 
00067   // Called when mongoose has finished processing request.
00068   void (*end_request)(const struct mg_connection *, int reply_status_code);
00069 
00070   // Called when mongoose is about to log a message. If callback returns
00071   // non-zero, mongoose does not log anything.
00072   int  (*log_message)(const struct mg_connection *, const char *message);
00073 
00074   // Called when mongoose initializes SSL library.
00075   int  (*init_ssl)(void *ssl_context, void *user_data);
00076 
00077   // Called when websocket request is received, before websocket handshake.
00078   // If callback returns 0, mongoose proceeds with handshake, otherwise
00079   // cinnection is closed immediately.
00080   int (*websocket_connect)(const struct mg_connection *);
00081 
00082   // Called when websocket handshake is successfully completed, and
00083   // connection is ready for data exchange.
00084   void (*websocket_ready)(struct mg_connection *);
00085 
00086   // Called when data frame has been received from the client.
00087   // Parameters:
00088   //    bits: first byte of the websocket frame, see websocket RFC at
00089   //          http://tools.ietf.org/html/rfc6455, section 5.2
00090   //    data, data_len: payload, with mask (if any) already applied.
00091   // Return value:
00092   //    non-0: keep this websocket connection opened.
00093   //    0:     close this websocket connection.
00094   int  (*websocket_data)(struct mg_connection *, int bits,
00095                          char *data, size_t data_len);
00096 
00097   // Called when mongoose tries to open a file. Used to intercept file open
00098   // calls, and serve file data from memory instead.
00099   // Parameters:
00100   //    path:     Full path to the file to open.
00101   //    data_len: Placeholder for the file size, if file is served from memory.
00102   // Return value:
00103   //    NULL: do not serve file from memory, proceed with normal file open.
00104   //    non-NULL: pointer to the file contents in memory. data_len must be

00106   const char * (*open_file)(const struct mg_connection *,

00128 // Start web server.
00129 //
00130 // Parameters:
00131 //   callbacks: mg_callbacks structure with user-defined callbacks.
00132 //   options: NULL terminated list of option_name, option_value pairs that
00133 //            specify Mongoose configuration parameters.
00134 //
00135 // Side-effects: on UNIX, ignores SIGCHLD and SIGPIPE signals. If custom
00136 //    processing is required for these, signal handlers must be set up
00137 //    after calling mg_start().
00138 //
00139 //
00140 // Example:
00141 //   const char *options[] = {
00142 //     "document_root", "/var/www",
00143 //     "listening_ports", "80,443s",
00144 //     NULL
00145 //   };
00146 //   struct mg_context *ctx = mg_start(&my_func, NULL, options);
00147 //
00148 // Refer to https://github.com/valenok/mongoose/blob/master/UserManual.md
00149 // for the list of valid option and their possible values.
00150 //
00151 // Return:
00152 //   web server context, or NULL on error.
